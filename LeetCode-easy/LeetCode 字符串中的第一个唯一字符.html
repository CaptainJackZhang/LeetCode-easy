思路：两边循环，第一遍生成出现的次数，第二遍寻找第一个唯一的字符，效率有一点点低

代码：
class Solution(object):
    def firstUniqChar(self, s):
        """
        :type s: str
        :rtype: int
        """
        dic = {}
        for alpha in s:
            if dic.has_key(alpha):
                dic[alpha] += 1
            else:
                dic[alpha] = 1
        for i in s:
            if dic[i] == 1:
                return s.find(i)
        return -1
